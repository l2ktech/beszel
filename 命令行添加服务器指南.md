## 🐍 方式二：使用 Python 脚本

### 脚本代码

```python
#!/usr/bin/env python3
"""
Beszel 服务器添加脚本 - Python 版本
"""

import requests
import uuid
import json
from datetime import datetime

BASE_URL = "http://localhost:38005/api"
EMAIL = "442333521@qq.com"
PASSWORD = "!Wangzeyu66!@#"

def login():
    """登录并获取 Token"""
    response = requests.post(
        f"{BASE_URL}/collections/users/auth-with-password",
        json={
            "identity": EMAIL,
            "password": PASSWORD
        }
    )
    data = response.json()
    return data.get('token')

def get_user_id(token):
    """获取用户 ID"""
    headers = {"Authorization": f"Bearer {token}"}
    response = requests.get(
        f"{BASE_URL}/collections/users/records",
        headers=headers
    )
    data = response.json()
    return data.get('items', [{}])[0].get('id')

def create_system(token, user_id, name, host, port):
    """创建系统记录"""
    headers = {"Authorization": f"Bearer {token}"}
    response = requests.post(
        f"{BASE_URL}/collections/systems/records",
        headers=headers,
        json={
            "name": name,
            "host": host,
            "port": str(port),
            "status": "down",
            "users": [user_id]
        }
    )
    data = response.json()
    return data.get('id')

def create_fingerprint(token, system_id, ssh_key):
    """创建指纹记录"""
    agent_token = str(uuid.uuid4())
    headers = {"Authorization": f"Bearer {token}"}
    response = requests.post(
        f"{BASE_URL}/collections/fingerprints/records",
        headers=headers,
        json={
            "system": system_id,
            "token": agent_token,
            "fingerprint": ssh_key,
            "updated": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }
    )
    return agent_token

def add_server(name, host, port, ssh_key=None):
    """添加服务器"""
    if ssh_key is None:
        ssh_key = "ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAIAq8t3HXdYTMp8kWBwUuCbuezv/i74jcGgTlWvSaKJVA"

    # 1. 登录
    print(f"🔐 登录 Beszel...")
    token = login()
    if not token:
        print("❌ 登录失败")
        return

    # 2. 获取用户 ID
    print(f"👤 获取用户 ID...")
    user_id = get_user_id(token)
    if not user_id:
        print("❌ 获取用户 ID 失败")
        return

    # 3. 创建系统
    print(f"🖥️ 创建系统: {name} ({host}:{port})")
    system_id = create_system(token, user_id, name, host, port)
    if not system_id:
        print("❌ 创建系统失败")
        return

    # 4. 创建指纹
    print(f"🔑 创建指纹...")
    agent_token = create_fingerprint(token, system_id, ssh_key)

    # 5. 输出配置
    print(f"\n✅ 服务器添加成功！")
    print(f"\n系统信息:")
    print(f"  ID: {system_id}")
    print(f"  名称: {name}")
    print(f"  主机: {host}:{port}")
    print(f"\nAgent 配置:")
    print(f"  Token: {agent_token}")
    print(f"  SSH Key: {ssh_key[:50]}...")

    # 6. 生成 docker-compose.yml
    print(f"\n📝 在目标服务器执行以下命令:")
    print(f"\n# 创建目录")
    print(f"mkdir -p ~/beszel-agent && cd ~/beszel-agent")
    print(f"\n# 创建 docker-compose.yml")
    docker_compose = f"""services:
  beszel-agent:
    image: henrygd/beszel-agent
    container_name: beszel-agent
    restart: unless-stopped
    network_mode: host
    volumes:
      - ./beszel_agent_data:/var/lib/beszel-agent
      - /var/run/docker.sock:/var/run/docker.sock:ro
    environment:
      PORT: 45876
      KEY: '{ssh_key}'
      TOKEN: '{agent_token}'
"""
    print(docker_compose)
    print(f"\n# 启动 Agent")
    print(f"docker compose up -d")

if __name__ == "__main__":
    import sys

    if len(sys.argv) < 4:
        print("用法: python add_server.py <名称> <主机> <端口> [SSH密钥]")
        print("\n示例:")
        print("  python add_server.py 本地服务器 localhost 45876")
        print("  python add_server.py 生产服务器 192.168.1.100 22")
        sys.exit(1)

    name = sys.argv[1]
    host = sys.argv[2]
    port = sys.argv[3]
    ssh_key = sys.argv[4] if len(sys.argv) > 4 else None

    add_server(name, host, port, ssh_key)
```

### 使用 Claude API

```python
#!/usr/bin/env python3
"""
使用 Claude API 自动化添加服务器到 Beszel
"""

import requests
import anthropic
import json

# 配置
BESZEL_URL = "http://localhost:38005/api"
BESZEL_EMAIL = "442333521@qq.com"
BESZEL_PASSWORD = "!Wangzeyu66!@#"
ANTHROPIC_API_KEY = "your-anthropic-api-key"

def beszel_login():
    """登录 Beszel"""
    response = requests.post(
        f"{BESZEL_URL}/collections/users/auth-with-password",
        json={"identity": BESZEL_EMAIL, "password": BESZEL_PASSWORD}
    )
    return response.json().get('token')

def claude_analyze_server(server_info):
    """使用 Claude 分析服务器信息"""
    client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

    prompt = f"""
    你是服务器监控专家。分析以下服务器信息，生成 Beszel 配置建议:

    服务器信息:
    - 名称: {server_info['name']}
    - IP: {server_info['ip']}
    - 用途: {server_info['purpose']}
    - 操作系统: {server_info['os']}
    - CPU: {server_info['cpu']}
    - 内存: {server_info['memory']}
    - 磁盘: {server_info['disk']}

    请生成:
    1. 建议的监控阈值（CPU、内存、磁盘）
    2. 告警规则建议
    3. Docker 监控建议（如果适用）

    以 JSON 格式返回，包含 thresholds, alerts, recommendations 三个字段。
    """

    message = client.messages.create(
        model="claude-3-opus-20240229",
        max_tokens=1024,
        messages=[
            {"role": "user", "content": prompt}
        ]
    )

    return json.loads(message.content[0].text)

def main():
    # 示例服务器列表
    servers = [
        {
            "name": "生产服务器",
            "ip": "192.168.1.100",
            "purpose": "Web 应用服务器",
            "os": "Ubuntu 22.04",
            "cpu": "4 cores",
            "memory": "16GB",
            "disk": "500GB SSD"
        }
    ]

    # 登录
    token = beszel_login()
    if not token:

    # 处理每个服务器
    for server in servers:

        suggestions = claude_analyze_server(server)
        print(f"建议: {suggestions}")

if __name__ == "__main__":
    main()
```

## 🛠️ 完整示例：批量添加服务器

```bash
#!/bin/bash

# 批量添加服务器脚本

SERVERS=(
  "生产服务器:192.168.1.100:22"
  "测试服务器:192.168.1.101:22"
  "开发服务器:192.168.1.102:22"
)

for server in "${SERVERS[@]}"; do
  IFS=':' read -r name host port <<< "$server"
  echo "添加服务器: $name ($host:$port)"
  bash add_system.sh "$name" "$host" "$port"
  echo "---"
done
```
